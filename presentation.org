* Swap
- A standard swap function
#+BEGIN_SRC
#include <stdio.h>

void swap(int *i, int *j) {
  int t = *i;
  *i = *j;
  *j = t;
}

void main() {
   int a = 23, b = 47;
   swap(&a, &b);
}
#+END_SRC
