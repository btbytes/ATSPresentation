* Swap
- A standard swap function
#+BEGIN_SRC
#include <stdio.h>

void swap(int *i, int *j) {
  int t = *i;
  *i = *j;
  *j = t;
}

void main() {
   int a = 23, b = 47;
   swap(&a, &b);
}
#+END_SRC

* Swap
- Include C code directly
#+BEGIN_EXAMPLE
%{^
#include <stdio.h>
void swap(int *i, int *j) {
  int t = *i;
  *i = *j;
  *j = t;
}

void swap_runner() {
  int a = 1, b = 2;
  swap(&a, &b);
  printf("After a: %d, b: %d\n", a, b);
}
%}

extern fun swap_runner():void = "ext#swap_runner"

implement main0 () = swap_runner()
#+END_EXAMPLE
